SELECT   PUNTOS_VOTACIONS.DIRECCION,   PUNTOS_VOTACIONS.ID,   IFNULL(PUNTOS_VOTACIONS.NOMBRE,'') NOMBRE,   CIUDADES.NOMBRE AS CIUDAD FROM   CIUDADES INNER JOIN PUNTOS_VOTACIONS ON   CIUDADES.ID = PUNTOS_VOTACIONS.ID_CIUDAD WHERE   PUNTOS_VOTACIONS.NOMBRE LIKE '%s$buscar$s%' OR PUNTOS_VOTACIONS.DIRECCION LIKE '%s$buscar$s%' OR CIUDADES.NOMBRE LIKE '%s$buscar$s%'
